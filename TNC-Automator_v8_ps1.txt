#Requires -Version 5

# Purpose: Testing network connection from source computer/server to target computer/server
# Logic: Sequential testing of DNS, ICMP, and TCP Port status with automated TraceRoute.
<#
.DESCRIPTION
This script is a modular diagnostic tool designed for automated network validation and troubleshooting within enterprise environments. It follows a structured execution flow that mirrors the OSI model, beginning with local environment baselining via Get-NetIPConfiguration followed by a Layer 7 name resolution check using Resolve-DnsName. The script then validates Layer 3 reachability using ICMP and Layer 4 availability through a targeted TCP port test via Test-NetConnection. A key technical feature is the conditional diagnostic escalation: the script evaluates the TcpTestSucceeded boolean property and only initiates a -TraceRoute if the port test fails, optimizing execution time. To ensure comprehensive audit logging, it utilizes stream redirection (3>&1 2>&1) to capture the Warning and Error streams into a dynamically named, timestamped text file, providing a complete forensic record of connectivity failures and routing hops.
#>
# Author: Matthew Hinton
# Script version: 8
# ==============================================================================
# ADJUSTABLE SETTINGS: Only edit the values inside the quotation marks below
# ==============================================================================

# 1. The name of this computer (Leave as $env:computername for automatic detection)
$SourceServerName = $env:computername

# 2. The name or IP address of the server you want to test 
$TargetServerName = "10.0.0.x"

# 3. The port number you want to test (e.g., '53' for DNS, '443' for HTTPS)
$TargetServerPort = '80'

# 4. Where the log files will be saved on this computer. 
# IMPORTANT: Ensure the path ends with a backslash \
$Dir = "D:\tools\logs\"

# ==============================================================================
# END OF ADJUSTABLE SETTINGS (Do not edit below this line unless you know code)
# ==============================================================================

# Ensure the log folder exists; if it is missing, this creates it automatically.
if (!(Test-Path $Dir)) { 
    New-Item -ItemType Directory -Path $Dir | Out-Null 
}

# Formatting date and time for a unique filename to prevent logs from being overwritten.
# Filename example: 20260504T2039402559_sourcecomputer_tnctest_report.txt
$CurrentDate     = Get-Date -Format yyyy.MM.dd
$CurrentTime     = Get-Date -UFormat %R
$CurrentDateTime = Get-Date -Format FileDateTime
$LogFileName1    = $CurrentDateTime + "_" + $SourceServerName + "_tnctest_report.txt"
$PathToLogFileName = Join-Path $Dir $LogFileName1

# --- Compatibility check ---
$SourceServerPowershellVersion = $PSVersionTable.PSVersion

# Check to make sure that version of PowerShell is at least 5.1 before preceding.

    If($PSVersionTable.PSVersion.Major -le 4) {
        Throw "This script has not been tested with version 4.0 or older of PowerShell.  Please execute this script from a system that has PowerShell 5.1 installed.  Windows 10/11/Server 2016 and newer has it installed by default."
        }

# --- Main Execution ---

# Setup the report header for the text file.
$Header = @"
*********************************************************
NETWORK TEST REPORT
Source Server: $SourceServerName
Source Powershell version: $SourceServerPowershellVersion
Target Server: $TargetServerName
Target Port:   $TargetServerPort
Report Date:   $CurrentDate
Report Time:   $CurrentTime
*********************************************************
"@
# Change the output file Encoding from default to utf8.  The text file was getting strange characters and 
# nulls when viewed with notepad other than server.  This seemed to resolve the issue so output file is viewable with notepad. 
$Header | Out-File -FilePath $PathToLogFileName -Encoding 'utf8'

# 0. Capture local IP and DNS settings to see the starting configuration of this server.
Write-Output "`n[0] Source IP configuration:" | Out-File -FilePath $PathToLogFileName -Append
Get-NetIPConfiguration | Out-File -FilePath $PathToLogFileName -Append

# 1. Check if the MID server can translate the target name into an IP address.
Write-Output "`n[1] Checking DNS Resolution..." | Out-File -FilePath $PathToLogFileName -Append
try {
    # If the name is not found, the 'catch' block below will record the failure.
    Resolve-DnsName -Name $TargetServerName -ErrorAction Stop | Out-File -FilePath $PathToLogFileName -Append
} catch {
    Write-Output "CRITICAL: DNS Resolution failed. The server name might be misspelled or DNS is down." | Out-File -FilePath $PathToLogFileName -Append
}

# 2. Test general reachability (the standard Ping test)
# 3>&1 2>&1 ensures that any console warnings are saved into the text file.
Write-Output "`n[2] Testing General Connectivity (TNC)..." | Out-File -FilePath $PathToLogFileName -Append
tnc $TargetServerName -InformationLevel "Detailed" 3>&1 2>&1 | Out-File -FilePath $PathToLogFileName -Append

# 3. Test if the specific application port is open and accepting traffic.
Write-Output "`n[3] Testing Connection to Port: $TargetServerPort..." | Out-File -FilePath $PathToLogFileName -Append
$testResult = Test-NetConnection -ComputerName $TargetServerName -Port $TargetServerPort -InformationLevel "Detailed" 3>&1 2>&1
$testResult | Out-File -FilePath $PathToLogFileName -Append

# 4. If the port test fails, run a Traceroute to find where the connection is dropping.
if ($testResult.TcpTestSucceeded -eq $false) {
    Write-Output "`n[!] TCP Port Test FAILED. Running Traceroute to find the network blockage..." | Out-File -FilePath $PathToLogFileName -Append
    Test-NetConnection -ComputerName $TargetServerName -TraceRoute | Out-File -FilePath $PathToLogFileName -Append
} else {
    Write-Output "`n[+] TCP Port Test SUCCESSFUL." | Out-File -FilePath $PathToLogFileName -Append
}

# Provide a final summary in the console for the person running the script.
Write-Output "`nScript run complete." 
Write-Output "The report is saved at: $PathToLogFileName"
